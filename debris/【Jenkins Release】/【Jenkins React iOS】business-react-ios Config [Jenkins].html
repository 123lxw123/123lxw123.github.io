#指定项目的scheme名称
 
scheme="zaihui_crm"
 
#指定要打包的配置名
 
configuration=${BUILD_TYPE}
 
#指定打包所使用的输出方式，目前支持app-store, package, ad-hoc, enterprise, development, 和developer-id，即xcodebuild的method参数
 
if [ $BUILD_TYPE == "Release" ]
then
   export_method="app-store"
else
   export_method="development"
fi
 
#指定项目地址
 
workspace_path="$project_path/zaihui_crm.xcodeproj"
 
#指定输出路径
rm -rf "${project_path}/ipa"
 
mkdir -p "${project_path}/ipa"
 
output_path="${project_path}/ipa"
 
#指定输出归档文件地址
 
archive_path="$output_path/zaihui_crm_${export_method}_${now}.xcarchive"
 
#指定输出ipa地址
 
ipa_path="$output_path/zaihui_crm_${BUILD_TYPE}_${BUILD_BRANCH}_${now}.ipa"
 
#指定输出ipa名称
 
ipa_name="zaihui_crm_${export_method}_${BUILD_TYPE}_${now}.ipa"
 
#输出设定的变量值
 
echo "===workspace path: ${workspace_path}==="
 
echo "===archive path: ${archive_path}==="
 
echo "===ipa path: ${ipa_path}==="
 
echo "===export method: ${export_method}==="
